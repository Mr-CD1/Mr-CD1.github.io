python os模块

发表于 2019-12-01

常用的命令

创建文件夹
os.makedir('dir1/dir2') #可生成多层递归目录,可以创建不存在的中间文件夹；若dir1不存在，则自己创建dir1
os.mkdir('dir') #生成单级目录
os.mkdir('dir1/dir2') #若dir1不存在则返回错误
只能删空文件夹
os.rmdir('dir1,/dir2') #删除单级空目录，若目录不为空则无法删除
os.removedirs('dir1/dir2') #若目录为空则删除，并递归到上一级目录，若仍为空，则删除
os.listdir('dirname') #相当于shell中的ls
os.remove() #删除一个文件
os.remove("oldname","newname") #重命名文件
os.stat('path/filename') #获取文件/目录信息
os.chdir('dirname') #改变当前工作目录
os.getcwd() #确定当前工作目录
os.environ(key) #查看相关的环境变量
os.popen(command,mode,bufsize) #从一个命令中打开一个管道
    command:使用的命令
    mode:权限
    bufsize:指明文件需要的缓冲大小：0-无缓冲；1-行缓冲
os.system("bash command") #运行shell命令，直接显示
os.umask() #查看umask值，控制默认权限
os.getuid() #获得操作系统的用户ID
os.access(path,mode) #使用当前的UID尝试访问路径。
    参数：path：要用来检测是否有访问权限的路径；
        mode:1.os.F-OK:路径是否存在
             2.os.R_OK:是否可读
             3.os._OK:是否可写
             4.os.X_OK:是否可执行
os.walk()

os.path模块

os.path.abspath(path) #获取文件的绝对路径
os.path.basename(path) #获取文件名称
os.path.commonprefix(list) #返回list中，path共有的最长路径    
os.path.dirname(filename) #返回文件路径
os.path.exists(path) #判断路径是否存在，若存在则返回TRUE
os.path.getatime(path) #返回文件最近访问的时间
os.path.getmtime(path) #返回文件最近修改的时间
os.path.getctime(path) #返回文件创建的时间
os.path.join(path1,path2,...) #拼接路径
os.path.splitext(path) #将文件名和扩展名分开
os.path.split(path) #返回文件的路径和文件名
